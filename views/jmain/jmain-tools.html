  • fnExtend()

    Nos permite combinar multiples objetos, manteniendo referencias a los objetos originales.

    var extremidadSuperior = {
	hombro: {},
	brazo: {},
	codo: {},
	antebrazo: {},
	muneca: {},
	mano: {}
}; // end 

var extremidadInferior = {
	ingle: {},
	cadera: {},
	muslo: {},
	rodilla: {},
	pierna: {},
	tobillo: {},
	pie: {}
}

var cuerpo = {};
fnExtend(cuerpo, extremidadSuperior, extremidadInferior);
console.log(cuerpo);
