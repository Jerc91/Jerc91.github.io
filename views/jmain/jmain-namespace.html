• Espacio de Nombres

  • Assembly()

    Nuestro código puede estar mejor estructurado con espacios de nombres, así podemos dividir funcionalidades dentro de la aplicación. Hay diferente formas de crearlos, por ejemplo: 

    var myApp = {};
myApp.tools = {};
myApp.tools.fnPrint = function (text) {
	console.log(text);
};
myApp.tools.fnPrint('¡Hola Luna!');

    La mejor forma para crear espacio de nombres es dentro de una única función, dentro de jmain.js se encuentra la función Assembly la que cual crea un espacio de nombre, por cada instancia de ésta función tendrá una función fnAddNS para crear un nuevo espacio de nombre, la cual recive como parametro el nombre. 

    var myApp = new Assembly();
myApp.fnAddNS('tools');
myApp.tools.fnPrint = function (text) {
	console.log(text);
};
myApp.tools.fnPrint('¡Hola Luna!');

    Otra caracteristica de usar espacios de nombre, es encapsular tanto variables como métodos, para esto se puede usar el patrón de diseño modular, para la creación de esta técnica se usa una función autoejecutable. 

    var myApp = new Assembly();
(function () {
	this.fnPrint = function (text) {
		console.log(text);
	};
}).call(myApp.fnAddNS('tools'));
myApp.tools.fnPrint('¡Hola Luna!');

  • fnImport()

    Busca en la variable windows el espacio de nombres ingresado como parámetro, se retornará el objecto que coincida con el valor buscado. 

    var myApp = new Assembly();
(function () {
	this.fnPrint = function (text) {
		console.log(text);
	}; // end function
}).call(myApp.fnAddNS('tools'));

(function() {
	var fnPrint = fnImport('myApp.tools.fnPrint');

	this.fnSum = function() {
		var total = 0;
		for(var i in arguments)
			total += arguments[i]; 
		
		fnPrint(total);
	}; // end function
}).call(myApp.fnAddNS('calc'));

myApp.calc.fnSum(1,1,1);

  • fnDepenciesTrue()

    Verifica si los elementos ingresados como parámetros son diferentes de: null, undefined y 0. Esto para verificar que los elementos importados sean validos. 

    var myApp = new Assembly();
/*
(function () {
	this.fnPrint = function (text) {
		console.log(text);
	}; // end function
}).call(myApp.fnAddNS('tools'));
*/

(function() {
	var fnPrint = fnImport('myApp.tools.fnPrint');
	if(!fnDepenciesTrue(fnPrint)) {
		console.log('Se requiere el nombre de espacio myApp.tools.fnPrint');
		return;
	} // end if

	this.fnSum = function() {
		var total = 0;
		for(var i in arguments)
			total += arguments[i]; 
		
		fnPrint(total);
	}; // end function
}).call(myApp.fnAddNS('calc'));
